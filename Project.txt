Q 4.1.4

Youngest Traveller
SELECT person.personname, person.age, city.name, city.population
FROM city
INNER JOIN hasvisitedcity on hasvisitedcity.cityID = city.ID
Inner JOIN person on person.personID = hasvisitedcity.personID
ORDER by age ASC
limit 1;
